/* Let's write the query to INNER join two tables*/
SELECT l.description AS left , r.description AS right
  FROM left As l
  JOIN right AS r ON l.id= r.id;

/*------Outer join ----*/
SELECT l.description AS left , r.description AS right
  FROM left As l
  LEFT JOIN right AS r ON l.id= r.id;

       s.date,
       i.name,
       i.description,
       s.price
  FROM sale AS s
       JOIN
       item AS i ON s.item_id = i.id;
